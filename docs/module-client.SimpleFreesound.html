Example
<script type="text/javascript" src="simple-freesound.min.js"></script>

<script type="text/javascript">
  var apiKey = "your_freesound_api_key_goes_here";
  var sf = new simpleFreesound.SimpleFreesound(apiKey);

  sf.query({
    search: [ 'drum', 'bass' ],
    duration: [ 0.01, 1 ]
  });
</script>
