import http from 'http';
import path from 'path';
import fs from 'fs';

import FreesoundQuery from '../common/FreesoundQuery';

const cwd = process.cwd();
const relativeFreesoundPath = path.join('sounds', 'freesound');
const freesoundPath = path.join(cwd, 'public', relativeFreesoundPath);

class FreesoundNode extends FreesoundQuery {
  constructor(apiKey) {
    super(apiKey);
  }

  /**
   * Download hq mp3 previews from current stored sound information.
   *
   * @param {Object} queryParams - The query parameters.
   * @param {Array.}
   * @param {String} destination - The downloaded files destination folder.
   */
  download(queryParams, destination = '.') {
    return new Promise((resolve, reject) => {
      super.query(queryParams)
        .then(updatedIds => this._downloadFilesFromUrls(updatedIds, destination))
        .then(() => resolve(this.currentSndsInfo))
        .catch(error => console.error(error.stack));
    });
  }

  /** @private */
  _downloadFilesFromUrls(ids, destination) {
    const promises = [];

    for (let id of ids)
      promises.push(this._downloadFileFromUrl(id, destination));

    return Promise.all(promises);
  }

  /** @private */
  _downloadFileFromUrl(id, destination) {
    return new Promise((resolve, reject) => {
      const dst = path.join(freesoundPath, `${id}.mp3`)
      const file = fs.createWriteStream(dst);
      const request = http.get(
        this.sndsInfo.get(id)['previews']['preview-hq-mp3'],
        response => {
          response.pipe(file);

          file.on('finish', () => {
            // in Freesound
            const url = path.join(relativeFreesoundPath, `${id}.mp3`);
            this.sndsInfo.get(id).url = url;
            this.currentSndsInfo.get(id).url = url;
            file.close();
            resolve();
          });

          file.on('error', err => {
            fs.unlink(dst);
            throw new Error(`Error downloading file ${id} : ${err}`);
          });
        }
      );
    });
  }

  /**
   * Retrieve sound information list from JSON file.
   *
   * @param {String} filename - Url of the file to read.
   */
  readFromFile(filename) {
    var si = JSON.parse(fs.readFileSync(filename, 'utf-8'));
    if (si) {
      this.sndsInfo = new Map();

      for (let i in si)
        this.sndsInfo.set(si[i]['id'], si[i]);
    }
  }

  /**
   * Dump sound information list to JSON file.
   *
   * @param {String} filename - Url of the file to dump the list writeToFile.
   */
  writeToFile(filename) {
    let output = [];

    for (let i of this.sndsInfo.keys())
      output.push(this.sndsInfo.get(i));

    fs.writeFileSync(filename, JSON.stringify(output, null, 2), 'utf-8');
  }
};

export default FreesoundNode;
