Example
import SimpleFreesound from 'simple-freesound';

const fs = new SimpleFreesound('myApiKey', './downloads');
fs.query({
  search: [ 'space', 'insect' ],
  duration: [ 1, 20 ],
})
.then(() => fs.download())
.then(() => {
  console.log(fs.currentSoundsInfo);
});
